import requests
import random
import string
import time

# --- CONFIGURATION ---
WEBHOOK_URL = "https://discord.com/api/webhooks/1500243297950765178/9Y4nQfvGODB-7dq21foPEZ1qaDKUApFCZ6yEu23wvMW1QbqlvbiWQBnjjvGmSL6g70He"
ROBLOX_VALIDATE_URL = "https://auth.roblox.com/v1/usernames/validate"
LETTERS = string.ascii_lowercase
CHECK_INTERVAL = 100

class Roblox5LChecker:
    def __init__(self):
        self.session = requests.Session()
        self.csrf_token = None
        self.checked_count = 0
        self.valid_usernames = []
        self._refresh_csrf()

    def _refresh_csrf(self):
        """Fetches the initial CSRF token required for the Roblox API."""
        try:
            req = self.session.post("https://auth.roblox.com/v2/login")
            if "x-csrf-token" in req.headers:
                self.csrf_token = req.headers["x-csrf-token"]
                self.session.headers.update({"x-csrf-token": self.csrf_token})
                print("[!] CSRF Token acquired successfully.")
        except Exception as e:
            print(f"[X] Failed to get CSRF token: {e}")

    def generate_username(self):
        """Generates a random 5-letter username."""
        return ''.join(random.choice(LETTERS) for _ in range(5))

    def check_username(self, username):
        """Checks if the username is available using the signup endpoint."""
        payload = {
            "birthday": "2000-01-01T00:00:00.000Z",
            "context": "Signup",
            "username": username
        }
        try:
            res = self.session.post(ROBLOX_VALIDATE_URL, json=payload)
            
            # If CSRF is invalid or expired, refresh and retry
            if res.status_code == 403 and "x-csrf-token" in res.headers:
                self._refresh_csrf()
                return self.check_username(username)
                
            # 429 means Rate Limited (Too Many Requests)
            if res.status_code == 429:
                print("[!] Rate limited by Roblox! Sleeping for 5 seconds...")
                time.sleep(5)
                return False

            data = res.json()
            # Code 0 typically means the username is valid and available
            if data.get("code") == 0:
                return True
            return False
            
        except Exception as e:
            return False

    def send_webhook(self):
        """Sends a stylish embed message to the Discord Webhook."""
        description = f"**Total Checked:** `{self.checked_count}`\n\n"
        
        if self.valid_usernames:
            # Prefixes the found usernames with @
            formatted_names = "\n".join([f"> `@{u}`" for u in self.valid_usernames])
            description += f"**Available Usernames Found ({len(self.valid_usernames)}):**\n{formatted_names}"
        else:
            description += "**Available Usernames Found:**\n> `None yet.`"

        embed = {
            "title": "🔍 Roblox 5L Checker Status",
            "color": 5763719, # Greenish color
            "description": description,
            "thumbnail": {
                "url": "https://upload.wikimedia.org/wikipedia/commons/thumb/3/3a/Roblox_player_icon_black.svg/512px-Roblox_player_icon_black.svg.png"
            },
            "footer": {
                "text": "Automated Termux Checker • Update every 100 checks"
            }
        }

        try:
            requests.post(WEBHOOK_URL, json={"embeds": [embed]})
            print(f"[~] Webhook sent! (Checked: {self.checked_count})")
        except Exception as e:
            print(f"[X] Webhook sending failed: {e}")

    def run(self):
        print("========================================")
        print("      Roblox 5L Username Checker        ")
        print("========================================")
        print("[+] Starting the checker process...\n")
        
        while True:
            username = self.generate_username()
            is_valid = self.check_username(username)
            self.checked_count += 1
            
            if is_valid:
                print(f"[+] AVAILABLE : @{username}")
                self.valid_usernames.append(username)
            else:
                print(f"[-] TAKEN     : {username}")
                
            # Send webhook notification every 100 checks
            if self.checked_count % CHECK_INTERVAL == 0:
                self.send_webhook()
                
            # Anti-Rate Limit Delay (crucial for keeping the IP unbanned)
            time.sleep(0.3)

if __name__ == "__main__":
    bot = Roblox5LChecker()
    bot.run()
