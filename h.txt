import requests
import random
import string
import time
import threading
from concurrent.futures import ThreadPoolExecutor

WEBHOOK_URL = "https://discord.com/api/webhooks/1500243297950765178/9Y4nQfvGODB-7dq21foPEZ1qaDKUApFCZ6yEu23wvMW1QbqlvbiWQBnjjvGmSL6g70He"
ROBLOX_VALIDATE_URL = "https://auth.roblox.com/v1/usernames/validate"
THREAD_COUNT = 5
CHECK_INTERVAL = 100
LETTERS = string.ascii_lowercase

checked_count = 0
valid_usernames = []
csrf_token = ""
lock = threading.Lock()
last_request_time = 0
REQUEST_DELAY = 0.5

def generate_username():
    return "".join(random.choice(LETTERS) for _ in range(5))

def get_csrf(session):
    global csrf_token
    for attempt in range(3):
        try:
            r = session.post(
                "https://auth.roblox.com/v2/login",
                timeout=30,
                json={"ctype": "Username", "cvalue": "test", "password": "test"}
            )
            token = r.headers.get("x-csrf-token", "")
            if token:
                csrf_token = token
                print(f"[CSRF] ✓ Token acquired: {token[:20]}...")
                return True
        except requests.exceptions.Timeout:
            print(f"[CSRF] Timeout attempt {attempt + 1}/3, retrying...")
            time.sleep(2)
        except Exception as e:
            print(f"[CSRF] Error attempt {attempt + 1}/3: {str(e)[:50]}")
            time.sleep(2)
    
    print("[CSRF] ✗ Failed after 3 attempts")
    return False

def send_webhook(session):
    global valid_usernames, checked_count

    with lock:
        total_checked = checked_count
        names = valid_usernames.copy()
        valid_usernames.clear()

    if names:
        found_text = "\n".join([f"`@{name}`" for name in names[:10]])
        if len(names) > 10:
            found_text += f"\n...and {len(names) - 10} more"
    else:
        found_text = "None found in this batch"

    embed = {
        "title": "🔍 Osusma Username Scanner",
        "color": 0x57F287,
        "description": (
            f"**Status:** ✅ Active\n"
            f"**Total Checked:** `{total_checked}`\n"
            f"**Available in This Batch:** `{len(names)}`\n\n"
            f"**Latest Available:**\n{found_text}"
        ),
        "footer": {
            "text": "Scanning: 5-letter usernames • Letters only"
        }
    }

    payload = {
        "username": "Osusma Scanner",
        "embeds": [embed]
    }

    try:
        response = session.post(WEBHOOK_URL, json=payload, timeout=15)
        if response.status_code == 204:
            print(f"[WEBHOOK] ✓ Update sent - Total: {total_checked}")
        else:
            print(f"[WEBHOOK] ✗ Failed with status: {response.status_code}")
    except Exception as e:
        print(f"[WEBHOOK] ✗ Error: {str(e)[:50]}")

def check_username():
    global checked_count, csrf_token, last_request_time

    session = requests.Session()
    session.headers.update({
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36",
        "Accept": "application/json, text/plain, */*",
        "Accept-Language": "en-US,en;q=0.9",
        "Origin": "https://www.roblox.com",
        "Referer": "https://www.roblox.com/"
    })
    
    # CSRF token al - 5 saniye bekle
    time.sleep(2)
    while not csrf_token:
        if get_csrf(session):
            break
        print("[WAIT] CSRF token not available, waiting 5 seconds...")
        time.sleep(5)

    error_count = 0
    max_errors = 10

    while True:
        try:
            # Rate limiting
            with lock:
                current_time = time.time()
                time_since_last = current_time - last_request_time
                if time_since_last < REQUEST_DELAY:
                    time.sleep(REQUEST_DELAY - time_since_last)
                last_request_time = time.time()

            # CSRF token yoksa yenile
            if not csrf_token:
                print("[CSRF] Token lost, refreshing...")
                if not get_csrf(session):
                    time.sleep(10)
                    continue

            username = generate_username()

            payload = {
                "birthday": "2000-01-01T00:00:00.000Z",
                "context": "Signup",
                "username": username
            }

            headers = {
                "x-csrf-token": csrf_token,
                "Content-Type": "application/json"
            }

            res = session.post(
                ROBLOX_VALIDATE_URL, 
                json=payload, 
                headers=headers, 
                timeout=30
            )

            # CSRF token yenileme
            if res.status_code == 403:
                if "x-csrf-token" in res.headers:
                    with lock:
                        csrf_token = res.headers["x-csrf-token"]
                    print(f"[CSRF] ✓ Token refreshed from 403 response")
                    continue
                else:
                    print("[403] Possible rate limit, waiting 5 seconds...")
                    time.sleep(5)
                    get_csrf(session)
                    continue

            # Rate limiting
            if res.status_code == 429:
                retry_after = res.headers.get("Retry-After", "5")
                try:
                    wait_time = int(retry_after)
                except:
                    wait_time = 5
                print(f"[429] Rate limited, waiting {wait_time}s...")
                time.sleep(wait_time)
                continue

            if res.status_code == 200:
                error_count = 0
                try:
                    data = res.json()
                    is_valid = data.get("code") == 0

                    with lock:
                        checked_count += 1
                        current_count = checked_count

                        if is_valid:
                            print(f"[✓] AVAILABLE: @{username}")
                            valid_usernames.append(username)
                        else:
                            print(f"[✗] TAKEN:     @{username}")

                        if current_count % CHECK_INTERVAL == 0:
                            send_webhook(session)
                except:
                    print(f"[ERROR] Failed to parse response for {username}")
                    continue
            else:
                print(f"[{res.status_code}] Unexpected status for {username}")
                error_count += 1

        except requests.exceptions.Timeout:
            print(f"[TIMEOUT] Request timed out for {username if 'username' in locals() else 'unknown'}")
            error_count += 1
        except requests.exceptions.ConnectionError as e:
            print(f"[CONNECTION] Error: {str(e)[:50]}")
            time.sleep(5)
            error_count += 1
        except Exception as e:
            print(f"[ERROR] {str(e)[:80]}")
            error_count += 1

        # Hata limiti kontrolü
        if error_count >= max_errors:
            print(f"[PAUSE] {error_count} errors, waiting 30s and refreshing CSRF...")
            time.sleep(30)
            error_count = 0
            get_csrf(session)

        time.sleep(0.2)

def run_checker():
    print("=" * 60)
    print("🤖 Roblox 5-Letter Username Scanner")
    print("=" * 60)
    print(f"⚙️  Configuration:")
    print(f"   • Threads: {THREAD_COUNT}")
    print(f"   • Webhook updates: Every {CHECK_INTERVAL} checks")
    print(f"   • Request delay: {REQUEST_DELAY}s between requests")
    print(f"   • Username: 5 letters (a-z)")
    print(f"   • Timeout: 30 seconds")
    print("=" * 60)
    print("[START] Initializing scanner...\n")
    
    try:
        with ThreadPoolExecutor(max_workers=THREAD_COUNT) as executor:
            futures = []
            for i in range(THREAD_COUNT):
                futures.append(executor.submit(check_username))
                print(f"[THREAD] Worker #{i+1} started")
                time.sleep(1)  # Thread'leri yavaş başlat
            
            print("\n[READY] All workers active - scanning usernames...\n")
            
            for future in futures:
                future.result()
                
    except KeyboardInterrupt:
        print("\n\n[STOP] Scanner terminated by user")
    except Exception as e:
        print(f"\n[FATAL] Critical error: {e}")

if __name__ == "__main__":
    run_checker()
