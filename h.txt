import requests
import random
import string
import time
import threading
from concurrent.futures import ThreadPoolExecutor

WEBHOOK_URL = "https://discord.com/api/webhooks/1500243297950765178/9Y4nQfvGODB-7dq21foPEZ1qaDKUApFCZ6yEu23wvMW1QbqlvbiWQBnjjvGmSL6g70He"
ROBLOX_VALIDATE_URL = "https://auth.roblox.com/v1/usernames/validate"
THREAD_COUNT = 5
CHECK_INTERVAL = 100
LETTERS = string.ascii_lowercase

checked_count = 0
valid_usernames = []
csrf_token = ""
lock = threading.Lock()

def generate_username():
    return "".join(random.choice(LETTERS) for _ in range(5))

def get_csrf(session):
    global csrf_token
    try:
        r = session.post("https://auth.roblox.com/v2/login", timeout=10)
        token = r.headers.get("x-csrf-token", "")
        if token:
            csrf_token = token
    except:
        pass

def send_webhook(session):
    global valid_usernames, checked_count

    with lock:
        total_checked = checked_count
        names = valid_usernames[:]
        valid_usernames.clear()

    if names:
        found_text = "\n".join([f"`@{name}`" for name in names[:15]])
        if len(names) > 15:
            found_text += f"\n...and {len(names) - 15} more"
    else:
        found_text = "None"

    embed = {
        "title": "Roblox Username Checker",
        "color": 5763719,
        "description": (
            f"**Status:** Active\n"
            f"**Checked:** `{total_checked}`\n"
            f"**Available Found:** `{len(names)}`\n\n"
            f"**Latest Results:**\n{found_text}"
        ),
        "footer": {
            "text": "Automated Username Scan"
        }
    }

    payload = {
        "username": "Username Checker",
        "embeds": [embed]
    }

    try:
        session.post(WEBHOOK_URL, json=payload, timeout=10)
    except:
        pass

def check_username():
    global checked_count, csrf_token

    session = requests.Session()
    get_csrf(session)

    while True:
        username = generate_username()
        payload = {
            "birthday": "2000-01-01T00:00:00.000Z",
            "context": "Signup",
            "username": username
        }

        headers = {"x-csrf-token": csrf_token}

        try:
            res = session.post(ROBLOX_VALIDATE_URL, json=payload, headers=headers, timeout=10)

            if res.status_code == 403 and "x-csrf-token" in res.headers:
                with lock:
                    csrf_token = res.headers["x-csrf-token"]
                continue

            if res.status_code == 429:
                time.sleep(2)
                continue

            data = res.json()
            is_valid = data.get("code") == 0

            with lock:
                checked_count += 1
                current_count = checked_count

                if is_valid:
                    print(f"[+] AVAILABLE : @{username}")
                    valid_usernames.append(username)
                else:
                    print(f"[-] TAKEN     : {username}")

                if current_count % CHECK_INTERVAL == 0:
                    send_webhook(session)

        except:
            pass

def run_checker():
    print("Roblox 5-letter username checker started")
    with ThreadPoolExecutor(max_workers=THREAD_COUNT) as executor:
        for _ in range(THREAD_COUNT):
            executor.submit(check_username)

if __name__ == "__main__":
    run_checker()
