import asyncio
import aiohttp
import random
import string

WEBHOOK_URL = "https://discord.com/api/webhooks/1500243297950765178/9Y4nQfvGODB-7dq21foPEZ1qaDKUApFCZ6yEu23wvMW1QbqlvbiWQBnjjvGmSL6g70He"
ROBLOX_VALIDATE_URL = "https://auth.roblox.com/v1/usernames/validate"
CHARS = string.ascii_lowercase + string.digits
THREAD_COUNT = 30
CHECK_INTERVAL = 100

checked_count = 0
valid_usernames = []
csrf_token = ""

async def get_csrf(session):
    async with session.post("https://auth.roblox.com/v2/login") as resp:
        return resp.headers.get("x-csrf-token")

async def check_health(session, token):
    test_name = "".join(random.choice(CHARS) for _ in range(25))
    payload = {
        "birthday": "2000-01-01T00:00:00.000Z",
        "context": "Signup",
        "username": test_name
    }
    headers = {"x-csrf-token": token}
    async with session.post(ROBLOX_VALIDATE_URL, json=payload, headers=headers) as resp:
        if resp.status == 200:
            data = await resp.json()
            return data.get("code") == 0
        return False

async def send_webhook(total, valids):
    if not valids:
        return
    formatted = "\n".join([f"> `@{u}`" for u in valids])
    embed = {
        "title": "🔍 Roblox Checker Status",
        "color": 2829617,
        "description": f"**Total Checked:** `{total}`\n\n**Found Usernames:**\n{formatted}",
        "footer": {"text": "Async High-Speed System"}
    }
    async with aiohttp.ClientSession() as session:
        await session.post(WEBHOOK_URL, json={"embeds": [embed]})

async def worker(session):
    global checked_count, csrf_token, valid_usernames
    while True:
        username = "".join(random.choice(CHARS) for _ in range(5))
        if random.random() > 0.8:
            username = "".join(random.choice(CHARS) for _ in range(random.randint(5, 6)))
            
        headers = {"x-csrf-token": csrf_token}
        payload = {
            "birthday": "2000-01-01T00:00:00.000Z",
            "context": "Signup",
            "username": username
        }

        try:
            async with session.post(ROBLOX_VALIDATE_URL, json=payload, headers=headers) as resp:
                if resp.status == 403:
                    csrf_token = resp.headers.get("x-csrf-token")
                    continue
                if resp.status == 429:
                    await asyncio.sleep(12)
                    continue
                if resp.status == 200:
                    data = await resp.json()
                    checked_count += 1
                    if data.get("code") == 0:
                        print(f"[+] AVAILABLE: {username}")
                        valid_usernames.append(username)
                    if checked_count % CHECK_INTERVAL == 0:
                        asyncio.create_task(send_webhook(checked_count, list(valid_usernames)))
                        valid_usernames.clear()
        except:
            await asyncio.sleep(1)

async def main():
    global csrf_token
    async with aiohttp.ClientSession() as session:
        csrf_token = await get_csrf(session)
        print("[!] Running integrity check...")
        if not await check_health(session, csrf_token):
            print("[X] ERROR: API is returning false positives (IP Shadowbanned).")
            return
        print("[V] System verified. Starting tasks.\n")
        tasks = [asyncio.create_task(worker(session)) for _ in range(THREAD_COUNT)]
        await asyncio.gather(*tasks)

if __name__ == "__main__":
    asyncio.run(main())
