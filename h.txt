import requests
import re
import sys
import json
from collections import defaultdict
from getpass import getpass

# ─────────────────────────────────────────
#  Roblox Group Payout Analyzer
#  Kullanım: python roblox_payout_analyzer.py
# ─────────────────────────────────────────

GROUP_ID = 827675604

HEADERS_BASE = {
    "User-Agent": "Mozilla/5.0",
    "Accept": "application/json",
    "Content-Type": "application/json",
}

def get_session(cookie: str) -> requests.Session:
    session = requests.Session()
    session.cookies.set(".ROBLOSECURITY", cookie, domain=".roblox.com")
    r = session.post("https://auth.roblox.com/v2/logout")
    csrf = r.headers.get("x-csrf-token")
    if csrf:
        session.headers.update({"x-csrf-token": csrf})
    session.headers.update(HEADERS_BASE)
    return session

def verify_login(session: requests.Session):
    r = session.get("https://users.roblox.com/v1/users/authenticated")
    if r.status_code == 200:
        return r.json()
    return None

def fetch_audit_logs(session: requests.Session, group_id: int) -> list:
    logs = []
    cursor = ""
    page = 1

    print("\n📥 Audit loglar çekiliyor...")

    while True:
        url = (
            f"https://groups.roblox.com/v1/groups/{group_id}/audit-log"
            f"?actionType=SpendGroupFunds&limit=100"
        )
        if cursor:
            url += f"&cursor={cursor}"

        r = session.get(url)

        if r.status_code == 403:
            print("❌ Erişim reddedildi. Hesabın bu grubun owner/admin'i değil olabilir.")
            break
        if r.status_code != 200:
            print(f"❌ API hatası: {r.status_code} - {r.text}")
            break

        data = r.json()
        items = data.get("data", [])
        logs.extend(items)
        print(f"   Sayfa {page}: {len(items)} kayıt alındı (toplam: {len(logs)})")

        cursor = data.get("nextPageCursor")
        if not cursor:
            break
        page += 1

    return logs

def extract_text(description) -> str:
    """description string veya dict olabilir, her ikisini de handle eder."""
    if isinstance(description, str):
        return description
    if isinstance(description, dict):
        # Roblox bazen {"recipient": "...", "amount": 50} gibi dict döner
        # Tüm değerleri string olarak birleştir
        parts = []
        for v in description.values():
            parts.append(str(v))
        return " ".join(parts)
    return str(description)

def parse_from_dict(description: dict):
    """
    Roblox API dict formatından direkt recipient ve amount çeker.
    Olası anahtarlar: recipient, amount, robux, username, displayName
    """
    recipient = None
    amount = 0

    # Recipient için olası anahtarlar
    for key in ["recipient", "username", "displayName", "name", "targetUser"]:
        if key in description:
            recipient = str(description[key])
            break

    # Amount için olası anahtarlar
    for key in ["amount", "robux", "robuxAmount", "value"]:
        if key in description:
            try:
                amount = int(description[key])
            except (ValueError, TypeError):
                pass
            break

    return recipient, amount

def analyze_logs(logs: list) -> dict:
    totals = defaultdict(int)
    unmatched = 0

    for entry in logs:
        description = entry.get("description", "")
        actor = entry.get("actor", {})
        actor_name = actor.get("user", {}).get("username", "Bilinmeyen")

        # ── Dict formatı ──
        if isinstance(description, dict):
            recipient, amount = parse_from_dict(description)
            if recipient and amount > 0:
                totals[recipient] += amount
                continue
            # Dict'ten çekemedik, string'e çevir ve regex dene
            description_str = extract_text(description)
        else:
            description_str = description

        # ── String / Regex formatı ──
        # Format 1: "payout of Robux from group funds to UserName (50)"
        m = re.search(r'payout of Robux from group funds to ([^\(]+)\s*\((\d+)\)', description_str)
        if m:
            totals[m.group(1).strip()] += int(m.group(2))
            continue

        # Format 2: "UserName spent 50 of group funds"
        m2 = re.search(r'(\w+) spent (\d+) of group funds', description_str)
        if m2:
            totals[m2.group(1).strip()] += int(m2.group(2))
            continue

        # Format 3: Sadece sayı varsa actor'a yaz
        m3 = re.search(r'\b(\d+)\b', description_str)
        if m3:
            totals[actor_name] += int(m3.group(1))
            continue

        unmatched += 1

    if unmatched > 0:
        print(f"⚠️  {unmatched} kayıt hiçbir formata uymadı, atlandı.")

    return dict(totals)

def print_report(totals: dict):
    if not totals:
        print("\n⚠️  Hiç payout kaydı bulunamadı.")
        return

    print("\n" + "═" * 52)
    print("          💰 PAYOUT ANALİZ RAPORU")
    print("═" * 52)
    print(f"  Grup ID : {GROUP_ID}")
    print("─" * 52)
    print(f"  {'Kullanıcı Adı':<32} {'Toplam Robux':>12}")
    print("─" * 52)

    sorted_totals = sorted(totals.items(), key=lambda x: x[1], reverse=True)
    grand_total = 0

    for username, amount in sorted_totals:
        print(f"  {username:<32} {amount:>10,} R$")
        grand_total += amount

    print("─" * 52)
    print(f"  {'TOPLAM':<32} {grand_total:>10,} R$")
    print("═" * 52)
    print(f"\n  Toplam {len(totals)} farklı kişiye ödeme yapılmış.")

def save_report(totals: dict):
    filename = "payout_raporu.txt"
    try:
        with open(filename, "w", encoding="utf-8") as f:
            f.write(f"Grup ID: {GROUP_ID}\n")
            f.write("=" * 52 + "\n")
            f.write(f"{'Kullanıcı Adı':<32} {'Toplam Robux':>12}\n")
            f.write("-" * 52 + "\n")
            sorted_totals = sorted(totals.items(), key=lambda x: x[1], reverse=True)
            grand_total = 0
            for username, amount in sorted_totals:
                f.write(f"{username:<32} {amount:>10,} R$\n")
                grand_total += amount
            f.write("-" * 52 + "\n")
            f.write(f"{'TOPLAM':<32} {grand_total:>10,} R$\n")
        print(f"\n💾 Rapor '{filename}' dosyasına kaydedildi.")
    except Exception as e:
        print(f"⚠️  Dosya kaydedilemedi: {e}")

def debug_first_log(logs: list):
    """İlk kaydın ham halini gösterir — format anlamak için."""
    if logs:
        print("\n🔍 İlk log kaydı (ham format):")
        print(json.dumps(logs[0], indent=2, ensure_ascii=False))

def main():
    print("╔════════════════════════════════════════════╗")
    print("║     Roblox Group Payout Analyzer           ║")
    print("║     Grup ID: 827675604                     ║")
    print("╚════════════════════════════════════════════╝\n")

    print("🔐 .ROBLOSECURITY cookie'nizi girin:")
    print("   (Roblox.com > F12 > Application > Cookies > .ROBLOSECURITY)\n")
    cookie = getpass("Cookie: ").strip()

    if not cookie:
        print("❌ Cookie boş olamaz.")
        sys.exit(1)

    print("\n🔄 Giriş yapılıyor...")
    session = get_session(cookie)
    user = verify_login(session)

    if not user:
        print("❌ Giriş başarısız. Cookie geçersiz veya süresi dolmuş.")
        sys.exit(1)

    print(f"✅ Giriş başarılı! Hoş geldin, {user.get('name', '?')} (ID: {user.get('id', '?')})")

    logs = fetch_audit_logs(session, GROUP_ID)

    if not logs:
        print("\n⚠️  Log bulunamadı veya erişim yok.")
        sys.exit(1)

    print(f"\n✅ Toplam {len(logs)} payout kaydı bulundu. Analiz ediliyor...")

    # İlk kaydı debug olarak göster
    debug_first_log(logs)

    totals = analyze_logs(logs)
    print_report(totals)

    save_q = input("\n📄 Raporu .txt dosyasına kaydetmek ister misin? (e/h): ").strip().lower()
    if save_q == "e":
        save_report(totals)

    print("\n✅ İşlem tamamlandı.")

if __name__ == "__main__":
    main()
