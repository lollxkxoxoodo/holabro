import aiohttp
import asyncio
from collections import defaultdict

# --- YAPILANDIRMA (AYARLAR) ---
GROUP_ID = 827675604
COOKIE = "BURAYA_.ROBLOSECURITY_COOKIENIZI_YAPISTIRIN"

# DRY_RUN True iken kimseye gerçekten rol verilmez, sadece konsola yazdırılır.
# Sonuçlardan emin olduktan sonra False yapın.
DRY_RUN = True 

# Rollerin gereksinimleri (Satın alma sayısı)
ROLE_THRESHOLDS = {
    "Topline": 15,
    "Elite": 10,
    "Plus": 5
}

class RobloxGroupBot:
    def __init__(self, group_id, cookie):
        self.group_id = group_id
        self.cookie = cookie
        self.headers = {
            "Cookie": f".ROBLOSECURITY={self.cookie}",
            "Accept": "application/json"
        }
        self.csrf_token = None
        self.role_ids = {} # Rol isimlerini ID'leri ile eşleştirmek için

    async def get_csrf_token(self, session):
        """Roblox'tan CSRF token alır."""
        url = "https://auth.roblox.com/v2/logout"
        async with session.post(url, headers=self.headers) as response:
            self.csrf_token = response.headers.get("x-csrf-token")
            self.headers["X-CSRF-TOKEN"] = self.csrf_token
            if not self.csrf_token:
                print("[HATA] CSRF Token alınamadı. Cookie'nizin geçerli olduğundan emin olun.")

    async def safe_request(self, session, method, url, **kwargs):
        """CSRF süresi dolarsa otomatik yenileyen güvenli istek fonksiyonu."""
        if not self.csrf_token:
            await self.get_csrf_token(session)

        kwargs['headers'] = self.headers
        async with session.request(method, url, **kwargs) as response:
            if response.status == 403 and "x-csrf-token" in response.headers:
                # Token süresi dolmuş, yenisini alıp tekrar deniyoruz
                self.csrf_token = response.headers.get("x-csrf-token")
                self.headers["X-CSRF-TOKEN"] = self.csrf_token
                kwargs['headers'] = self.headers
                return await session.request(method, url, **kwargs)
            
            # Rate limit (Çok fazla istek) koruması
            if response.status == 429:
                print("[UYARI] Rate limit'e takıldık. 5 saniye bekleniyor...")
                await asyncio.sleep(5)
                return await self.safe_request(session, method, url, **kwargs)
                
            return response

    async def fetch_group_roles(self, session):
        """Gruptaki rolleri ve ID'lerini çeker."""
        print("[1/4] Grubun rolleri alınıyor...")
        url = f"https://groups.roblox.com/v1/groups/{self.group_id}/roles"
        
        response = await self.safe_request(session, 'GET', url)
        if response.status == 200:
            data = await response.json()
            for role in data.get("roles", []):
                self.role_ids[role["name"]] = role["id"]
        else:
            print(f"[HATA] Roller çekilemedi. Durum Kodu: {response.status}")

    async def fetch_group_members(self, session):
        """Gruptaki üyeleri ve mevcut rollerini çeker."""
        print("[2/4] Grup üyeleri taranıyor (Bu işlem üye sayısına göre sürebilir)...")
        members = {}
        url = f"https://groups.roblox.com/v1/groups/{self.group_id}/users?limit=100"
        cursor = ""

        while cursor is not None:
            paged_url = f"{url}&cursor={cursor}" if cursor else url
            response = await self.safe_request(session, 'GET', paged_url)
            
            if response.status == 200:
                data = await response.json()
                for item in data.get("data", []):
                    user_id = item["user"]["userId"]
                    role_name = item["role"]["name"]
                    members[user_id] = {"username": item["user"]["username"], "role": role_name}
                cursor = data.get("nextPageCursor")
            else:
                print(f"[HATA] Üyeler çekilemedi. Durum: {response.status}")
                break
        return members

    async def fetch_sales_data(self, session):
        """Satış geçmişini tarayarak kimin kaç eşya aldığını hesaplar."""
        print("[3/4] Satış geçmişi taranıyor...")
        sales_count = defaultdict(int)
        url = f"https://economy.roblox.com/v2/groups/{self.group_id}/transactions?transactionType=Sale&limit=100"
        cursor = ""

        while cursor is not None:
            paged_url = f"{url}&cursor={cursor}" if cursor else url
            response = await self.safe_request(session, 'GET', paged_url)
            
            if response.status == 200:
                data = await response.json()
                for transaction in data.get("data", []):
                    # Sadece kullanıcıların yaptığı alışverişleri say (grubun kendi kendine yaptığı vb. hariç)
                    if transaction.get("agent", {}).get("agentType") == "User":
                        user_id = transaction["agent"]["id"]
                        sales_count[user_id] += 1
                cursor = data.get("nextPageCursor")
            else:
                print(f"[HATA] Satışlar çekilirken sorun oluştu. Durum: {response.status}")
                break
                
        print(f"  → Toplam {sum(sales_count.values())} satış işlemi analiz edildi.")
        return sales_count

    async def change_user_role(self, session, user_id, username, new_role_name):
        """Kullanıcının rolünü günceller."""
        new_role_id = self.role_ids.get(new_role_name)
        if not new_role_id:
            print(f"[HATA] '{new_role_name}' isimli rolün ID'si bulunamadı!")
            return False

        if DRY_RUN:
            print(f"  [DRY_RUN] {username} (ID: {user_id}) -> {new_role_name} rolüne YÜKSELTİLECEK.")
            return True

        url = f"https://groups.roblox.com/v1/groups/{self.group_id}/users/{user_id}"
        payload = {"roleId": new_role_id}
        
        response = await self.safe_request(session, 'PATCH', url, json=payload)
        if response.status == 200:
            print(f"  [BAŞARILI] {username} -> {new_role_name} rolüne yükseltildi.")
            return True
        else:
            print(f"  [BAŞARISIZ] {username} için rol verilemedi. Kod: {response.status}")
            return False

    def determine_deserved_role(self, purchase_count):
        """Satın alma sayısına göre hak edilen rolü belirler."""
        if purchase_count >= ROLE_THRESHOLDS["Topline"]: return "Topline"
        if purchase_count >= ROLE_THRESHOLDS["Elite"]: return "Elite"
        if purchase_count >= ROLE_THRESHOLDS["Plus"]: return "Plus"
        return None

    def get_role_hierarchy(self):
        """Rollerin hiyerarşik sırasını döner (Düşük rolü yüksekle ezmemek için)."""
        return {"Guest": 0, "Plus": 1, "Elite": 2, "Topline": 3}

    async def run(self):
        async with aiohttp.ClientSession() as session:
            print("Oturum açılıyor ve API bağlantıları kuruluyor...")
            await self.fetch_group_roles(session)
            
            if not self.role_ids:
                print("[KRİTİK HATA] Rol bilgileri alınamadı. Bot durduruluyor.")
                return

            members = await self.fetch_group_members(session)
            sales = await self.fetch_sales_data(session)

            print("\n[4/4] Analiz ve Rol Dağıtımı Başlıyor...")
            if DRY_RUN:
                print("────────────────────────────────────────────────────────────")
                print("  *** DRY RUN AKTİF — Gerçek rol verilmeyecek ***")
                print("────────────────────────────────────────────────────────────")

            hierarchy = self.get_role_hierarchy()
            updates_made = 0

            for user_id, purchase_count in sales.items():
                if user_id not in members:
                    # Kullanıcı grupta değil, atla.
                    continue
                
                user_info = members[user_id]
                current_role = user_info["role"]
                deserved_role = self.determine_deserved_role(purchase_count)

                # Eğer kullanıcının hak ettiği bir rol varsa
                if deserved_role:
                    current_rank = hierarchy.get(current_role, -1)
                    deserved_rank = hierarchy.get(deserved_role, -1)

                    # Sadece mevcut rolden daha yüksek bir rol hak ediyorsa güncelle
                    # (Yetkilileri, Owner'ı veya dev'leri düşürmemek için hiyerarşi kullanıyoruz)
                    if deserved_rank > current_rank and current_rank != -1:
                        success = await self.change_user_role(session, user_id, user_info["username"], deserved_role)
                        if success:
                            updates_made += 1
                            # API'yi yormamak için gerçek işlemlerde ufak bir gecikme
                            if not DRY_RUN:
                                await asyncio.sleep(0.5)

            if updates_made == 0:
                print("\nVerilecek veya güncellenecek yeni rol bulunamadı.")
            else:
                print(f"\nToplam {updates_made} kullanıcı işlendi.")

if __name__ == "__main__":
    bot = RobloxGroupBot(GROUP_ID, COOKIE)
    # Windows'da asyncio hatalarını önlemek için:
    import sys
    if sys.platform == 'win32':
        asyncio.set_event_loop_policy(asyncio.WindowsSelectorEventLoopPolicy())
        
    asyncio.run(bot.run())
