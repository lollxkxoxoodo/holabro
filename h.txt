import requests
import random
import string
import time
from concurrent.futures import ThreadPoolExecutor
import threading

# --- CONFIGURATION ---
WEBHOOK_URL = "https://discord.com/api/webhooks/1500243297950765178/9Y4nQfvGODB-7dq21foPEZ1qaDKUApFCZ6yEu23wvMW1QbqlvbiWQBnjjvGmSL6g70He"
ROBLOX_VALIDATE_URL = "https://auth.roblox.com/v1/usernames/validate"
LETTERS = string.ascii_lowercase
CHECK_INTERVAL = 100
THREAD_COUNT = 15  # Adjust this to change the speed (15 threads = ~15-20 checks/sec depending on ping)

# --- GLOBAL VARIABLES ---
checked_count = 0
valid_usernames = []
csrf_token = ""
lock = threading.Lock()

def get_csrf():
    """Fetches the initial CSRF token required for the Roblox API."""
    global csrf_token
    try:
        req = requests.post("https://auth.roblox.com/v2/login")
        if "x-csrf-token" in req.headers:
            csrf_token = req.headers["x-csrf-token"]
            print("[!] CSRF Token acquired successfully.")
    except Exception as e:
        print(f"[X] Failed to get CSRF token: {e}")

def generate_username():
    """Generates a random 5-letter username."""
    return ''.join(random.choice(LETTERS) for _ in range(5))

def send_webhook():
    """Sends a stylish embed message to the Discord Webhook."""
    global valid_usernames
    
    with lock:
        current_checked = checked_count
        current_valids = list(valid_usernames)
        valid_usernames.clear() # Clear the list after sending

    description = f"**Total Checked:** `{current_checked}`\n\n"
    
    if current_valids:
        formatted_names = "\n".join([f"> `@{u}`" for u in current_valids])
        description += f"**Available Usernames Found ({len(current_valids)}):**\n{formatted_names}"
    else:
        description += "**Available Usernames Found:**\n> `None yet.`"

    embed = {
        "title": "🔍 Roblox 5L Checker Status",
        "color": 5763719,
        "description": description,
        "thumbnail": {
            "url": "https://upload.wikimedia.org/wikipedia/commons/thumb/3/3a/Roblox_player_icon_black.svg/512px-Roblox_player_icon_black.svg.png"
        },
        "footer": {
            "text": "Automated System • Status Update"
        }
    }

    try:
        requests.post(WEBHOOK_URL, json={"embeds": [embed]})
        print(f"\n[~] Webhook sent successfully! (Checked: {current_checked})\n")
    except Exception as e:
        print(f"\n[X] Webhook sending failed: {e}\n")

def check_username():
    """Worker function for threads to check usernames."""
    global checked_count, csrf_token
    
    while True:
        username = generate_username()
        payload = {
            "birthday": "2000-01-01T00:00:00.000Z",
            "context": "Signup",
            "username": username
        }
        headers = {"x-csrf-token": csrf_token}

        try:
            res = requests.post(ROBLOX_VALIDATE_URL, json=payload, headers=headers)
            
            # Refresh CSRF if expired
            if res.status_code == 403 and "x-csrf-token" in res.headers:
                with lock:
                    csrf_token = res.headers["x-csrf-token"]
                continue
                
            # Rate Limit check
            if res.status_code == 429:
                print(f"[!] RATE LIMITED! Thread pausing for 5 seconds... (IP restriction)")
                time.sleep(5)
                continue

            data = res.json()
            is_valid = (data.get("code") == 0)

            with lock:
                checked_count += 1
                current_count = checked_count
                
                if is_valid:
                    print(f"[+] AVAILABLE : @{username}")
                    valid_usernames.append(username)
                else:
                    print(f"[-] TAKEN     : {username}")
                
                # Send webhook notification every 100 checks
                if current_count % CHECK_INTERVAL == 0:
                    send_webhook()

        except Exception as e:
            pass

def run_checker():
    print("========================================")
    print("      Roblox 5L Username Checker        ")
    print("      [High Speed Threaded Mode]        ")
    print("========================================")
    
    get_csrf()
    print(f"[+] Starting {THREAD_COUNT} threads...\n")
    
    # Start threads
    with ThreadPoolExecutor(max_workers=THREAD_COUNT) as executor:
        for _ in range(THREAD_COUNT):
            executor.submit(check_username)

if __name__ == "__main__":
    run_checker()
