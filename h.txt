import aiohttp
import asyncio
import sys
from collections import defaultdict

# --- YAPILANDIRMA ---
GROUP_ID = 827675604
COOKIE = "BURAYA_.ROBLOSECURITY_DEGERINI_YAPISTIRIN"
DRY_RUN = True  # İlk başta True kalsın, test edin.

ROLE_THRESHOLDS = {
    "Topline": 15,
    "Elite": 10,
    "Plus": 5
}

class TermuxRobloxBot:
    def __init__(self, group_id, cookie):
        self.group_id = group_id
        self.cookie = cookie
        # Termux/Mobil cihaz gibi görünmek için User-Agent ekledik
        self.headers = {
            "Cookie": f".ROBLOSECURITY={self.cookie}",
            "Accept": "application/json",
            "User-Agent": "Mozilla/5.0 (Linux; Android 10; K) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0.0.0 Mobile Safari/537.36"
        }
        self.csrf_token = None
        self.role_ids = {}

    async def get_csrf_token(self, session):
        """Bağlantı kopmalarına karşı dirençli CSRF alıcı"""
        url = "https://auth.roblox.com/v2/logout"
        try:
            async with session.post(url, headers=self.headers) as response:
                token = response.headers.get("x-csrf-token")
                if token:
                    self.csrf_token = token
                    self.headers["X-CSRF-TOKEN"] = token
                    return True
        except Exception as e:
            print(f"[!] CSRF Alınırken Bağlantı Koptu, Tekrar Deneniyor... {e}")
            await asyncio.sleep(2)
            return await self.get_csrf_token(session)
        return False

    async def safe_request(self, session, method, url, **kwargs):
        """Bağlantı hatalarını (Connection Closed) yakalayan güvenli istek motoru"""
        if not self.csrf_token:
            await self.get_csrf_token(session)

        kwargs['headers'] = self.headers
        
        try:
            async with session.request(method, url, **kwargs) as response:
                # CSRF Yenileme
                if response.status == 403:
                    token = response.headers.get("x-csrf-token")
                    if token:
                        self.csrf_token = token
                        self.headers["X-CSRF-TOKEN"] = token
                        return await self.safe_request(session, method, url, **kwargs)
                
                # Hız Sınırı
                if response.status == 429:
                    print("[!] Çok hızlı gidiyoruz, 10 saniye mola...")
                    await asyncio.sleep(10)
                    return await self.safe_request(session, method, url, **kwargs)
                
                if response.status == 200:
                    return await response.json()
                return None

        except (aiohttp.ClientConnectionError, aiohttp.ServerDisconnectedError):
            print("[!] Bağlantı koptu, 3 saniye sonra otomatik devam edilecek...")
            await asyncio.sleep(3)
            return await self.safe_request(session, method, url, **kwargs)

    async def run(self):
        # Termux'ta bağlantı havuzunu (connector) limitli tutmak daha sağlıklıdır
        connector = aiohttp.TCPConnector(limit_per_host=5)
        async with aiohttp.ClientSession(connector=connector) as session:
            print(f"\n--- [ {GROUP_ID} ] İÇİN ANALİZ BAŞLADI ---")
            
            # 1. Rolleri Al
            print("[1/4] Rol yapılandırması çekiliyor...")
            roles_data = await self.safe_request(session, 'GET', f"https://groups.roblox.com/v1/groups/{self.group_id}/roles")
            if not roles_data:
                print("Hata: Rol listesi alınamadı. Cookie veya internetinizi kontrol edin.")
                return
            
            for r in roles_data['roles']:
                self.role_ids[r['name']] = r['id']

            # 2. Üyeleri Al
            print("[2/4] Grup üyeleri listeleniyor...")
            members = {}
            cursor = ""
            while cursor is not None:
                url = f"https://groups.roblox.com/v1/groups/{self.group_id}/users?limit=100" + (f"&cursor={cursor}" if cursor else "")
                data = await self.safe_request(session, 'GET', url)
                if not data: break
                for m in data['data']:
                    members[m['user']['userId']] = {"name": m['user']['username'], "role": m['role']['name']}
                cursor = data.get("nextPageCursor")

            # 3. Satışları Al
            print("[3/4] Satış geçmişi taranıyor (Bu biraz sürebilir)...")
            sales = defaultdict(int)
            cursor = ""
            while cursor is not None:
                url = f"https://economy.roblox.com/v2/groups/{self.group_id}/transactions?transactionType=Sale&limit=100" + (f"&cursor={cursor}" if cursor else "")
                data = await self.safe_request(session, 'GET', url)
                if not data: break
                for t in data['data']:
                    if t.get('agent', {}).get('agentType') == 'User':
                        sales[t['agent']['id']] += 1
                cursor = data.get("nextPageCursor")
            
            # 4. Analiz ve Uygulama
            print(f"\n[4/4] Analiz Tamamlandı. Toplam {len(sales)} alıcı bulundu.")
            print("-" * 50)
            
            hierarchy = {"Guest": 0, "Plus": 1, "Elite": 2, "Topline": 3}
            
            for uid, count in sales.items():
                if uid in members:
                    m_info = members[uid]
                    curr_role = m_info['role']
                    
                    # Hak edilen rolü bul
                    deserved = None
                    if count >= ROLE_THRESHOLDS["Topline"]: deserved = "Topline"
                    elif count >= ROLE_THRESHOLDS["Elite"]: deserved = "Elite"
                    elif count >= ROLE_THRESHOLDS["Plus"]: deserved = "Plus"
                    
                    if deserved:
                        c_rank = hierarchy.get(curr_role, -1)
                        d_rank = hierarchy.get(deserved, -1)
                        
                        if d_rank > c_rank and c_rank != -1:
                            if DRY_RUN:
                                print(f"[DRY-RUN] {m_info['name']} ({count} Satın Alma): {curr_role} -> {deserved}")
                            else:
                                # Rol Değiştirme
                                patch_url = f"https://groups.roblox.com/v1/groups/{self.group_id}/users/{uid}"
                                res = await self.safe_request(session, 'PATCH', patch_url, json={"roleId": self.role_ids[deserved]})
                                if res is not None or res == {}: # Roblox bazen boş dict döner
                                    print(f"[BAŞARILI] {m_info['name']} artık {deserved}!")
                                await asyncio.sleep(0.5) # Termux'ta aşırı ısınma/ban önleyici

if __name__ == "__main__":
    bot = TermuxRobloxBot(GROUP_ID, COOKIE)
    try:
        asyncio.run(bot.run())
    except KeyboardInterrupt:
        print("\n[!] İşlem kullanıcı tarafından durduruldu.")
