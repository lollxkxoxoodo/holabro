import asyncio
import aiohttp
import random
import os
import time
from aiohttp_socks import ProxyConnector

class DiscordProChecker:
    def __init__(self, user_file, proxy_file, worker_count=5):
        self.user_file = user_file
        self.proxy_file = proxy_file
        self.worker_count = worker_count
        self.queue = asyncio.Queue()
        self.proxies = self._load_list(self.proxy_file)
        self.headers = {
            "Accept": "*/*",
            "Accept-Language": "en-US,en;q=0.9",
            "Content-Type": "application/json",
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/121.0.0.0 Safari/537.36"
        }
        self.active_backoff = 0

    def _load_list(self, path):
        if os.path.exists(path):
            with open(path, "r") as f:
                return [line.strip() for line in f if line.strip()]
        return []

    async def worker(self, session):
        while not self.queue.empty():
            username = await self.queue.get()
            
            if self.active_backoff > time.time():
                wait_time = self.active_backoff - time.time()
                await asyncio.sleep(wait_time)

            proxy = random.choice(self.proxies)
            proxy_url = f"http://{proxy}"
            url = f"https://discord.com/api/v9/users/@me/is-unique?username={username}"

            try:
                async with session.get(url, proxy=proxy_url, timeout=10) as resp:
                    if resp.status == 200:
                        data = await resp.json()
                        if data.get("taken") is False:
                            print(f"[+] MEVCUT: {username}")
                            with open("available.txt", "a") as f:
                                f.write(f"{username}\n")
                        else:
                            print(f"[-] DOLU: {username}")
                        
                        await asyncio.sleep(random.uniform(1.2, 2.5))

                    elif resp.status == 429:
                        retry_after = (await resp.json()).get("retry_after", 5)
                        self.active_backoff = time.time() + retry_after
                        print(f"[!] LIMIT: {proxy} | {retry_after}s Bekleniyor...")
                        await self.queue.put(username) 
                        await asyncio.sleep(retry_after)

                    elif resp.status in [401, 403]:
                        if proxy in self.proxies: self.proxies.remove(proxy)

            except Exception:
                if proxy in self.proxies: self.proxies.remove(proxy)
            
            finally:
                self.queue.task_done()

    async def run(self):
        users = self._load_list(self.user_file)
        for u in users:
            await self.queue.put(u)

        conn = aiohttp.TCPConnector(limit_per_host=10)
        async with aiohttp.ClientSession(connector=conn, headers=self.headers) as session:
            workers = [asyncio.create_task(self.worker(session)) for _ in range(self.worker_count)]
            await asyncio.gather(*workers)

if __name__ == "__main__":
    checker = DiscordProChecker("usernames.txt", "proxies.txt", worker_count=3)
    asyncio.run(checker.run())
