local function checkGroupMembership(plr)
    for _, groupId in ipairs(GROUP_IDS) do
        local success, isInGroup = pcall(function()
            return plr:IsInGroup(groupId)
        end)
        
        if success and isInGroup then
            return true
        end
    end
    return false
end

local function onPlayerAdded(plr)
    wait(1)
    
    if checkGroupMembership(plr) then
        createPlayerCard(plr)
        
        if plr ~= player then
            showNotification(plr.Name, plr.DisplayName, plr.UserId, false)
        end
    end
end

local function onPlayerRemoving(plr)
    local card = contentFrame:FindFirstChild(plr.Name)
    if card then
        showNotification(plr.Name, plr.DisplayName, plr.UserId, true)
        card:Destroy()
    end
end

for _, plr in pairs(Players:GetPlayers()) do
    task.spawn(function()
        onPlayerAdded(plr)
    end)
end

Players.PlayerAdded:Connect(onPlayerAdded)
Players.PlayerRemoving:Connect(onPlayerRemoving)
